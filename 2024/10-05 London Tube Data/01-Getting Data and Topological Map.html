Final Code to Grab data

Grab Data and Wraggle to an Edge List


## https://github.com/lmmx/tubeulator
import pandas as pd
import polars as pl
import igraph as ig
import tubeulator

## lines?
lines = tubeulator.load_lines()

## get eveything but nationa-rail way
arrivable_line_names = lines.filter(pl.col("Name") != "national-rail")["Name"].unique().sort().to_list()
query= ",".join(arrivable_line_names)

## query all lines; clean result list into a record eventual pandas dataframe.
result_models = tubeulator.fetch.line.arrivals_by_ids(ids=query)
results = [rm.model_dump() for rm in result_models]
df = pl.DataFrame(results)


## create an edge and node data from transport
time_edge = (df
    # .select("StationName", "DestinationName", "VehicleId","ExpectedArrival", "TimeToLive")
    .with_columns(
        time_travel = pl.col("ExpectedArrival") -
        pl.col("ExpectedArrival")
        .shift(1)
        .over(order_by = pl.col("ExpectedArrival"), partition_by = pl.col("VehicleId")),
        prev_station = pl.col("StationName")
        .shift(1)
        .over(order_by = pl.col("ExpectedArrival"), partition_by = pl.col("VehicleId"))
    )
    .rename(
        {"prev_station":"from_station","StationName":"to_station"}
    )
    .select(
       "from_station","to_station","time_travel", "VehicleId", "ExpectedArrival", "ModeName"
    )
    .filter(pl.col("time_travel").is_not_null())
)

## filter down to overground only
overground_edge = time_edge.sql("select * from self where ModeName = 'overground'").to_pandas()


## swap names with integer ids
unique_station = pd.concat([
    overground_edge['from_station'],
    overground_edge['to_station']
]).unique()

## create a unique mapping directionary with name being key, value being integer
map_rev = pd.DataFrame(dict(name = pd.Series(unique_station))).to_dict().get('name')
mapping ={val:key for key,val in map_rev.items()}

## create an edge list from igraph
edge_list = overground_edge.replace(
    {'from_station':mapping, 'to_station':mapping}
) # !pip install rpy2
graph = ig.Graph.DataFrame(
    edge_list
)
graph.vs['name'] = [x[1] for x in map_rev.items()]

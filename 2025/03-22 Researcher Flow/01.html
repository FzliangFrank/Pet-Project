def with_store(func):
    ns = Path('_store') / (func.__name__ + '.pkl')
    def wraper(*arg,  **kwd):
        if ns.exists():
            print("loading from store")
            with open(ns, 'rb') as f:
                rs = pickle.load(f)
            return rs
        else:
            print("rerun function")
            rs = func(*arg, **kwd)
            with open(ns, 'wb') as f:
                pickle.dump(rs, f)
            return rs
    return wraper

def search_ror(query):
    """
    Search the ROR API with a specified query
    Args:
        query (str): Search term for organizations
    Returns:
        dict: JSON response from the API
    """
    url = "https://api.ror.org/organizations"
    params = {"query": query}
    response = requests.get(url, params=params)
    return response.json()

def map_search_ror(university_ranking, uplimit = None):
    searches_record = []
    for row in university_ranking.iter_rows():
        university_name = row[1]
        row_id = row[0]
        # print(university_name)
        assert isinstance(university_name,str)
        assert isinstance(row_id, int)
        if row_id <= uplimit:
            first_3result = search_ror(university_name)['items'][0:3]
            searches_record += [{'rank_id':row[0],'result':first_3result}]
        else:
            break
    return searches_record

import re
def _get_first_name(result):
    return [rs['result'][0]['name'] for rs in result]
def _get_frist_id(result):
    return [re.sub(r'https://ror.org/', '', rs['result'][0]['id']) for rs in result]
def _get_all_isni_id(result):
    return [rs['result'][0]['external_ids']['ISNI'] for rs in result]
def _get_relationship(result):
    relrs:list[list[dict]] = []
    for rs in result:
        relationships = rs['result'][0]['relationships']
        rel:list[dict] = []
        # print(len(relationships))
        for r in relationships:
            relat_label = r['label']
            relat_id = re.sub(r'https://ror.org/', '', r['id'])
            rel += [{'label':relat_label,'id':relat_id}]
        # print(rel)
        relrs.append(rel)
    return relrs
def _get_external_ids(result):

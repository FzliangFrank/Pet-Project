institute_avg_rank = (
    uni_info

institute_ranks = (
    institute_avg_rank
    .group_by(_.id)
    .aggregate(
          _.label.first().name('institute_label')
        , _['name'].collect().name('institutes')
        , _.avg_rank.mean().name('avg_rank')
        , _.avg_rank.max().name('lower_rank')
        , _.avg_rank.min().name('higher_rank')
    )
    .rename(ror_id = 'id')
)

pp_career_flow = (
    pp_career_progression
    .select('orcid','ror_id','start_date','institute_label', 'avg_rank')
    # place avg_rankings into a bracket?
    .mutate(
        ibis.cases(
              (_.avg_rank < 50, 'Top 50')
            , (_.avg_rank < 100, 'Top 100')
            , (_.avg_rank < 200, 'Top 200')
            , (_.avg_rank < 500, 'Top 500')
            , (_.avg_rank < 1000, 'Top 1000')
            , (None, 'Unknown')
        ).name('rank_bucket')
    )
    # .group_by(_.orcid)
    .mutate(
          prev_rank = _.rank_bucket.lag().over(order_by = 'start_date', group_by = 'orcid')
        , prev_date = _.start_date.lag().over(order_by = 'start_date', group_by = 'orcid')
    )
    .mutate(
          tennue = _.start_date.delta( _.prev_date, unit='month').coalesce(0)
    )
    .group_by(_.orcid)
    .mutate(
        n_change = _.count()
    )
    .mutate(
        ibis.cases(
          (_.n_change == 1, '1. SAME POSITION')
        , ((_.n_change > 1) & (_.tennue != 0), '2. CHANGED POSITION')
        , ((_.n_change > 1) & (_.tennue == 0), '3. CHANGED POSITION WITHIN SAME MONTH')
        , (None, 'UNKNOWN')
        ).name('change_type')
    )
)
pp_career_flow_tmp = conn.create_table('pp_career_flow', pp_career_flow, temp=True, overwrite=True)
pp_career_flow_delta = (
    pp_career_flow_tmp
    .filter(
        _.change_type == '2. CHANGED POSITION'
    )
    .select(
        _.prev_rank.name('rank_from'),
        _.rank_bucket.name('rank_to'),
        _.tennue.name('tennue'),
        _.orcid.name('orcid'),
        _.start_date.name('date'),
    )
    .mutate(
        _.date.truncate('year').name('year').year().name('year')
    )
    .mutate(
        # split year into decade
        ibis.cases(
            (_.year < 2010, 'before_2010'),
            (_.year >= 2010, 'after_2010'),
            (None, 'unknown')
        ).name('decade')
    )
)
pp_career_flow_agg = (
    pp_career_flow_delta
     .group_by(
        _.rank_from,
        _.rank_to,
        _.decade
    )
    .aggregate(
        _.count().name('n')
    )
)
pp_career_edge_list = pp_career_flow_agg.to_pandas()
pp_career_edge_list.to_parquet('pp_career_edge_list.parquet')

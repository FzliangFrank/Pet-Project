def my_transformation(m1, w):
    m3=m1.dot(w)
    return m3

def plot_transformation(m1, m3, ax=None):
    if ax is None:
        ax=plt.gca()
    for i in range(m1.shape[0]):
        x=[m1[i,0],m3[i,0]]
        y=[m1[i,1],m3[i,1]]
        ax.plot(x,y, 'k-',alpha=0.8,zorder=5)
    
    ax.scatter(m1[:,0],m1[:,1], zorder=6,alpha=1)
    ax.scatter(m3[:,0],m3[:,1],zorder=7, alpha=0.7)
